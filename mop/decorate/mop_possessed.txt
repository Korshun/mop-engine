Actor MOPPossessedMonster : PlayerPawn
{
	-PICKUP
	+NOSKIN
	+NOTARGET
	-FRIENDLY
	
	Species "PossessedByMOP"
	Player.SoundClass "PossessedByMOP"
	
	States
	{
	Death:
		TNT1 A 1
		Loop
	}
}

Actor MOPPossessedWeapon : Weapon
{
	+UNDROPPABLE
	+WEAPON.CHEATNOTWEAPON

	States
	{
	Ready:
		TNT1 A 1 A_WeaponReady
		Loop
	Select:
		TNT1 A 0 A_Raise
		Loop
	Deselect:
		TNT1 A 1 A_Lower
		Loop
	Fire:
		TNT1 A 0
		Goto Ready
	Spawn:
		TNT1 A -1
		Stop
	}
}
